%%%%%%%%The following code used to reduce a fraction %%%%%%%
\newcommand{\reducedfraction}[2]{%
\ifnum #2=0
  \text{~undefined~}
\else
  \pgfmathsetmacro{\fract}{#1/#2}
  \pgfmathtruncatemacro{\floorfract}{floor(#1/#2)}
  \pgfmathtruncatemacro{\reducednumabs}{abs(#1/gcd(#1,#2))}
  \pgfmathtruncatemacro{\reduceddenabs}{abs(#2/gcd(#1,#2))}
  \ifdim\fract pt=\floorfract pt
    \floorfract
  \else
    \ifdim\fract pt>0 pt
      \frac{\reducednumabs}{\reduceddenabs}
    \else
      \ifdim\fract pt<0pt
        -\frac{\reducednumabs}{\reduceddenabs}
      \fi
    \fi
  \fi
\fi
}
