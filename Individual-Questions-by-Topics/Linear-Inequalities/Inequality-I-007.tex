\pgfmathtruncatemacro{\a}{-4}
\pgfmathtruncatemacro{\b}{5}
\pgfmathtruncatemacro{\c}{3}

\pgfmathtruncatemacro{\d}{\a*\c-\b}
\pgfmathtruncatemacro{\e}{\d+\b}



Solve the inequality and write the solution set in interval notation.
\[ \a x-\b \geq \d  \]


\begin{solution}
\[\begin{split}
\a x-\b &\geq  \d\\
\a x&\geq \e\\
x&\leq \c
\end{split}
\]
The solution set is $(-\infty, \c]$.
\end{solution}
