%%%%%%%%%%%% Assume a>0, a>b %%%%%%%%%%%%

\pgfmathtruncatemacro{\a}{2}
\pgfmathtruncatemacro{\b}{-3}

\pgfmathtruncatemacro{\c}{-4}

\pgfmathtruncatemacro{\e}{\a-\b}
\pgfmathtruncatemacro{\d}{\a*(\c+1)-\b*\c}
\pgfmathtruncatemacro{\f}{\d-\a}



Solve the inequality and write the solution set in interval notation.
\ifnum\d>0
\[ \a (x+1)>\b x +\d \]
\begin{solution}
\[\begin{split}
\a (x+1)&>\b x +\d \\
\a x+\a&>\b x +\d \\
\e x&>\f \\
x&>\c
\end{split}
\]
The solution set is $(\c, \infty)$.
\end{solution}



\else
\[ \a (x+1)>\b x \d \]


\begin{solution}
\[\begin{split}
\a (x+1)&>\b x \d \\
\a x+\a&>\b x \d \\
\e x&>\f \\
x&>\c
\end{split}
\]
The solution set is $(\c, \infty)$.
\end{solution}
\fi
