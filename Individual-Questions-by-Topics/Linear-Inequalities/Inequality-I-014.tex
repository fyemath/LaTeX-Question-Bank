%%%%%%%%%%%% Assume a>0, a>b %%%%%%%%%%%%

\pgfmathtruncatemacro{\a}{3}
\pgfmathtruncatemacro{\b}{-6}

\pgfmathtruncatemacro{\c}{-1}

\pgfmathtruncatemacro{\e}{\b-\a}
\pgfmathtruncatemacro{\d}{(\b-\a)*\c}




Solve the inequality and write the solution set in interval notation.
\[ \d+\a x>\b x  \]


\begin{solution}
\[\begin{split}
\d+\a x &>  \b x \\
\d &>  \e x\\
\c &< x\\
x &> \c\\
\end{split}
\]
The solution set is $(\c, \infty)$.
\end{solution}
