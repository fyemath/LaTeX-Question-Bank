%%%%%%%%%%%% Assume a<0, %%%%%%%%%%%%

\pgfmathtruncatemacro{\a}{6}
\pgfmathtruncatemacro{\b}{-2}

\pgfmathtruncatemacro{\c}{-3}

\pgfmathtruncatemacro{\e}{\a-\b}
\pgfmathtruncatemacro{\d}{\a*(\c-1)-\b*\c}
\pgfmathtruncatemacro{\f}{\d+\a}



Solve the inequality and write the solution set in interval notation.
\ifnum\d>0
\[ \a (x-1)<\b x +\d \]
\begin{solution}
\[\begin{split}
\a (x-1)&<\b x +\d \\
\a x-\a&<\b x +\d \\
\e x&<\f \\
\ifnum\e>0
x&<\c\\
\else
x&>\c\\
\fi
\end{split}
\]
The solution set is \ifnum\e>0 $(-\infty, \c)$ \else $(\c, \infty)$\fi.
\end{solution}



\else
\[ \a (x-1)<\b x \d \]


\begin{solution}
\[\begin{split}
\a (x-1)&<\b x \d \\
\a x-\a&<\b x \d \\
\e x&<\f \\
\e x&<\f \\
\ifnum\e>0
x&<\c\\
\else
x&>\c\\
\fi
\end{split}
\]
The solution set is \ifnum\e>0 $(-\infty, \c)$\else $(\c, \infty)$\fi.
\end{solution}
\fi
