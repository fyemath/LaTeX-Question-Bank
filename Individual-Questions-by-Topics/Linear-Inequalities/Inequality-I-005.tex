\pgfmathtruncatemacro{\a}{-3}
\pgfmathtruncatemacro{\b}{2}
\pgfmathtruncatemacro{\c}{2}

\pgfmathtruncatemacro{\d}{\a*\c+\b}
\pgfmathtruncatemacro{\e}{\d-\b}



Solve the inequality and write the solution set in interval notation.
\[ \a x+\b \leq \d  \]


\begin{solution}
\[\begin{split}
\a x+\b &\leq  \d\\
\a x&\leq \e\\
x&\geq \c
\end{split}
\]
The solution set is $[\c, \infty)$.
\end{solution}
