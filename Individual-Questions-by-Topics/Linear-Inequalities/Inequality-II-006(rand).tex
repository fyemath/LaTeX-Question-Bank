%%%%%%%%%%%% Assume a<0, %%%%%%%%%%%%


\pgfmathtruncatemacro{\a}{random(2,7)}
\pgfmathtruncatemacro{\b}{random(8,2)-10}
\pgfmathtruncatemacro{\c}{random(1,5)}

%%%%%%% 
\pgfmathtruncatemacro{\result}{-3}




\pgfmathtruncatemacro{\d}{\a*(\result +\c)-\b*\result}
\pgfmathtruncatemacro{\e}{\a-\b}
\pgfmathtruncatemacro{\f}{\d-\a*\c}
\pgfmathtruncatemacro{\g}{\a*\c}



Solve the inequality and write the solution set in interval notation.

% \begin{ocg}{result layer}{1}{0}

\ifnum\d>0
\[ \a (x+\c)\leq \b x +\d.\]
\begin{solution}
\[\begin{split}
\a (x+\c)&\leq \b x +\d \\
\a x+\g&\leq \b x +\d \\
\e x&\leq \f \\
\ifnum\e>0
x&\leq \result \\
\else
x\geq \result \\
\fi
\end{split}
\]
The solution set is \ifnum\e>0 $(-\infty, \result ]$ \else  $[\result , \infty)$\fi.
\end{solution}



\else
\[ \a (x+\c)\leq \b x \d. \]

\begin{solution}
\[\begin{split}
\a (x+\c)&\leq \b x \d \\
\a x+\g&\leq \b x \d \\
\e x&\leq \f \\
\ifnum\e> 0
x&\leq \result \\
\else
x&\geq \result \\
\fi
\end{split}
\]
The solution set is \ifnum\e>0 $(-\infty, \result ]$ \else  $[\result , \infty)$\fi.
\end{solution}

\fi
