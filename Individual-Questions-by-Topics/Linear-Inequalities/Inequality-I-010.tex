%%%%%%%%%%%% Assume a<b, c<0 %%%%%%%%%%%%

\pgfmathtruncatemacro{\a}{-2}
\pgfmathtruncatemacro{\b}{4}

\pgfmathtruncatemacro{\c}{-1}

\pgfmathtruncatemacro{\e}{\a-\b}
\pgfmathtruncatemacro{\d}{(\a-\b)*\c}




Solve the inequality and write the solution set in interval notation.
\[ \a x< \b x + \d  \]


\begin{solution}
\[\begin{split}
\a x &< \b x + \d\\
\e x&< \d\\
x&> \c
\end{split}
\]
The solution set is $(\c,-\infty)$.
\end{solution}
