 \edef\n{\n}


  \pgfmathtruncatemacro{\a}{random(1,3)}
  \pgfmathtruncatemacro{\b}{random(2,4)}
  \pgfmathtruncatemacro{\c}{random(3,1)}

\pgfmathtruncatemacro{\nm}{\n^\m}
\pgfmathtruncatemacro{\ntwo}{2^\m}
  \pgfmathtruncatemacro{\ma}{\m*\a}
  \pgfmathtruncatemacro{\mb}{\m*\b}
  \pgfmathtruncatemacro{\mc}{\m*\c}

Simplify the radical expression.
\[
      \sqrt[\m]{-\frac{\invisibleoneincoef{\nm}\varx^{\ma}}{\ntwo\vary^{\mb}}}.
\]

\begin{solution}
\[
\sqrt[\m]{-\frac{\invisibleoneincoef{\nm}\varx^{\ma}}{\ntwo\vary^{\mb}}}
      =
\sqrt[\m]{
  \left(
    -\frac
      { \invisibleoneincoef{\m}\varx^{\invisibleoneinpower{\a}} }
      { 2\vary^{\invisibleoneinpower{\b}} }
  \right)^\m
}
=
-\frac{\invisibleoneincoef{\n}\varx^{\invisibleoneinpower{\a}}}{2\vary^{\invisibleoneinpower{\b}}}
.
\]

\end{solution}
