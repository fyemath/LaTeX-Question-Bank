  \pgfmathtruncatemacro{\a}{random(1,5)}

Write the radical expression as rational exponent (assume all variables are positive).
\[
  \ifnum\m=2
    \sqrt{\varx^{\invisibleoneinpower{\a}}\vary^{\b}}
  \else
      \sqrt[\m]{\varx^{\invisibleoneinpower{\a}}\vary^{\b}}
  \fi.
\]

\begin{solution}
\[
  \ifnum\m=2
    \sqrt{\varx^{\invisibleoneinpower{\a}}\vary^{\b}}
  \else
      \sqrt[\m]{\varx^{\invisibleoneinpower{\a}}\vary^{\b}}
  \fi
=
\varx^{\reducedfraction{\a}{\m}}\vary^{\reducedfraction{\b}{\m}}.
\]

\end{solution}
