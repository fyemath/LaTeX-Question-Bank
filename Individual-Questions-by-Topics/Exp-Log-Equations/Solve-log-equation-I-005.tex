\pgfmathtruncatemacro{\aa}{random(2,4)}
\pgfmathtruncatemacro{\bb}{random(2,3)}
\pgfmathtruncatemacro{\cc}{(\aa)^(\bb)}



Solve the logarithmic equation 
\[\log_x \cc = \bb\]


\begin{solution}
		\[\begin{split}
		\log_x \cc &= \bb\\
		x^{\bb} &=\cc\\
		x & =\ifnum\bb=2 \sqrt{\cc} \else \sqrt[\bb]{\cc}\fi =\aa
		\end{split}
		\]
\end{solution}
