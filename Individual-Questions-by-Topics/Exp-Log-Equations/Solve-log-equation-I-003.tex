\pgfmathtruncatemacro{\a}{random(2,3)}
\pgfmathtruncatemacro{\asq}{\a^2}
\pgfmathtruncatemacro{\b}{random(5,15)}
\pgfmathtruncatemacro{\ans}{\a^2+\b}
Solve the logarithmic equation 
\[\log_{\a} \sqrt{x-\b}=1.\]


\begin{solution}
		\[\begin{split}
		\log_{\a} \sqrt{x-\b}&=1\\
		\sqrt{x-\b}&=\a\\
		x-\b&=\a^2\\
		 x &=\ans.
		\end{split}
		\]
\end{solution}
