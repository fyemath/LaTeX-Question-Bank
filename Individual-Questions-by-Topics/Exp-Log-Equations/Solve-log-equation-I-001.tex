\pgfmathtruncatemacro{\a}{random(2,4)}
\pgfmathtruncatemacro{\b}{random(5,15)}
\pgfmathtruncatemacro{\c}{10-\a}
Solve the logarithmic equation 
\[\log(\a x+\b)-\log x=1.\]


\begin{solution}
		\[\begin{split}
		\log(\a x+\b)-\log x&=1\\
		\log(\frac{\a x+\b}{x})&=1\\
		\frac{\a x+\b}{x}&=10^1\\
		\a x+\b&=10x\\
		 \c x&=\b\\
		 x&=\reducedfraction{\b}{\c}.
		\end{split}
		\]
\end{solution}
