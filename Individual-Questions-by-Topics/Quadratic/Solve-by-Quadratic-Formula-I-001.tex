\pgfmathdeclarerandomlist{varx}{{x}{y}}
% \pgfmathdeclarerandomlist{varx}{{2}{3}}
\pgfmathrandomitem{\varx}{varx}
\edef\varx{\varx}

\pgfmathdeclarerandomlist{oddnumbers}{{1}{5}}
\pgfmathrandomitem{\b}{oddnumbers}
\edef\b{\b}

\pgfmathtruncatemacro{\a}{random(2,3)}
\pgfmathtruncatemacro{\twoa}{2*\a}
% \pgfmathtruncatemacro{\b}{random(1,3)}
\pgfmathtruncatemacro{\c}{random(1,3)}
\pgfmathtruncatemacro{\Dt}{\b^2+4*\a*\c}


Solve the following equation
\[
  -\a\varx^2+\b\varx+\c=0.
\]

\begin{solution}
Since $a=-\a$ and $b=\b$ and $c=\c$, the equation has to two solution
\[\varx=\frac{-\b\pm\sqrt{\b^2-4\cdot(-\a)\cdot\c}}{2\cdot(-\a)}=\frac{-\b\pm\sqrt{\Dt}}{-\twoa}.\]

\end{solution}
