\pgfmathdeclarerandomlist{varx}{{x}{y}}
% \pgfmathdeclarerandomlist{varx}{{2}{3}}
\pgfmathrandomitem{\varx}{varx}
\edef\varx{\varx}

\pgfmathtruncatemacro{\a}{random(2,3)}
\pgfmathdeclarerandomlist{oddnumbers}{{1}{3}{5}}
\pgfmathrandomitem{\h}{oddnumbers}
\edef\h{\h}
\pgfmathtruncatemacro{\b}{2*\a*\h}
\pgfmathtruncatemacro{\B}{2*\h}
\pgfmathtruncatemacro{\c}{1}
\pgfmathtruncatemacro{\k}{\a*\h^2+\c}
\pgfmathtruncatemacro{\ak}{\a*\k}

Solve the following equation \textbf{by completing the square}
\[
  \invisibleoneincoef{\a}\varx^2+\b\varx-\c=0.
\]

\begin{solution}
\[
  \begin{split}
  \invisibleoneincoef{\a}\varx^2+\b\varx-\c &=0\\
  \invisibleoneincoef{\a}(\varx+\h)^2-\k&=0\\
  (\varx+\h)^2&=\reducedfraction{\k}{\a}\\
  \varx+\h&=\pm\sqrt{\reducedfraction{\k}{\a}}\\
  \varx&=-\h\pm\frac{\sqrt{\ak}}{\a}.
  \end{split}
\]
\end{solution}
