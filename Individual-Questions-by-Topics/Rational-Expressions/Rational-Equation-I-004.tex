 \pgfmathtruncatemacro{\s}{random(1,5)}

 \pgfmathtruncatemacro{\A}{\a*\c}
 \pgfmathtruncatemacro{\C}{(-\b)*(\d)}
 \pgfmathtruncatemacro{\B}{\a*(\d)+(-\b)*(\c)}

\pgfmathtruncatemacro{\D}{\r*\c+\a*\s}
\pgfmathtruncatemacro{\E}{\r*\d+\s*(-\b)-\t}


% \pgfmathtruncatemacro{\bminusa}{(\a)*(\b)}

%%%r/(ax+b)-s/(ax+b)(cx+d)=x/(cx+d) Roots: r=1 or s=-(rd+s)/a

Solve the  rational equation
\[
   \frac{\r}{ \invisibleoneincoef{\a}\varx \constdisplay{-\b} }
  +
  \frac{\s}{\invisibleoneincoef{\c}\varx \constdisplay{\d}}
  =
  \frac{\t}{\invisibleoneincoef{\A}\varx^2 \plusminus{\B}\varx \constdisplay{\C}} 
  .
\]

\begin{solution}
\[
	\begin{split}
		\frac{\r}{ \invisibleoneincoef{\a}\varx \constdisplay{-\b} }
   +
   \frac{\s}{\invisibleoneincoef{\c}\varx \constdisplay{\d}}
   &= 
   \frac{\t}{\invisibleoneincoef{\A}\varx^2 \plusminus{\B}\varx \constdisplay{\C}}
\\
    \r(\invisibleoneincoef{\c}\varx \constdisplay{\d})+ \invisibleoneincoef{\s}(\invisibleoneincoef{\a}\varx \constdisplay{-\b})
    &=\t 
\\
    \D \varx \constdisplay{\E}  &=0\\
    \varx &=\reducedfraction{-\E}{\D}
	\end{split}
\]
\end{solution}
