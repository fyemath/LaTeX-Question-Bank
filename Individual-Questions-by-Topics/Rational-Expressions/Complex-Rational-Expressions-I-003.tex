%%%%[a/(x+1)-1]/[b/(x+1)+1]=[(a-1)-x]/[x+(b+1)]

Simplify the rational expression.
\[
  \dfrac
  { 
    \d-\frac{\a}{\varx+\c}
  }
  {
    \d+\frac{\b}{\varx+\c}
  }
  .
\]

\begin{solution}
\[
	\begin{split}
  \dfrac
  {
    \frac{\invisibleoneincoef{\d}(\varx+\c)-\a}{\varx+\c}
  }
  {
    \frac{\invisibleoneincoef{\d}(\varx+\c)+\b}{\varx+\c}
  }  
  =&
     \dfrac
  {
    {\invisibleoneincoef{\d}(\varx+\c)-\a}
  }
  {
    {\invisibleoneincoef{\d}(\varx+\c)+\b}
  } 
\\
	=	&
    \dfrac
    {\invisibleoneincoef{\d}\varx \ifnum\A=0 \else \constdisplay{\A}\fi }
    {\invisibleoneincoef{\d}\varx \ifnum\B=0 \else \constdisplay{\B}\fi} 
	\end{split}
\]

\end{solution}
