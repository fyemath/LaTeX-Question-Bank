 \fi
  \edef\d{\d}


Simplify the rational expression.
\[
  \dfrac
  {
    \frac{\a}{\varx}-\frac{\b}{\varx^2}
  }
  {
    \frac{\c}{\varx}+\frac{\d}{\varx^2}
  }  
  .
\]

\begin{solution}
\[
	\begin{split}
\dfrac
  {
    \frac{\a}{\varx}-\frac{\b}{\varx^2}
  }
  {
    \frac{\c}{\varx}+\frac{\d}{\varx^2}
  } 
  =&
 \dfrac
  {
    \frac{\a\varx-\b}{\varx^2}
  }
  {
    \frac{\c\varx+\d}{\varx^2}
  } 
\\
	=	&
    \dfrac{\a\varx-\b}{\varx^2}
    \cdot
    \dfrac{\varx^2}{\c\varx+\d}
\\
	=	&
  \dfrac
      {\a\varx-\b}
      {\c\varx+\d}  
	\end{split}
\]

\end{solution}
