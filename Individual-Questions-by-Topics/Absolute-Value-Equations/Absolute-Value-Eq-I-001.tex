\pgfmathtruncatemacro{\a}{3*random(1,3)} 
\pgfmathtruncatemacro{\b}{3*random(3,5)} 

\pgfmathtruncatemacro{\aplusb}{\a+\b} 
\pgfmathtruncatemacro{\aminusb}{\a-\b} 

\pgfmathtruncatemacro{\rone}{(\a+\b)/3} 
\pgfmathtruncatemacro{\rtwo}{(\a-\b)/3} 

Solve the absolute value equation 
\[|3x-\a|=\b.\]

\begin{solution}

% First, check if the equation is $|X|=c$ form. If not, isolate the absolute value expression $|X|$.

% \centerline{$|\a-3x|=\b$}
% The equation is equivalent to 
% \[|3x-\a|=\b.\]
Since $\b>0$, this equation is equivalent to the following two equations.
\vspace*{-3em}
\begin{center}
	\begin{alignat*}{3}
		3x-\a&=\b &&\ctc{ or } & 3x-\a&=-\b\\
		3x&=\aplusb &&  & 3x&=\aminusb\\
		x&=\rone &&\ctc{or}  &  x&=\rtwo
	\end{alignat*}
\end{center}\vspace*{-1em} 
So the solutions are $x=\rone$ and $x=\rtwo$. In set-builder notation, the solution set is $\{\rone, \rtwo\}$.
\end{solution}
