\pgfmathtruncatemacro{\a}{random(2,5)}



 %%%%%% generate random b coprime with a %%%
 \ifcase\a\relax%
  \or \pgfmathrandomitem{\b}{}
  \or \pgfmathrandomitem{\b}{CoprimeWithTwo}
  \or \pgfmathrandomitem{\b}{CoprimeWithThree}
  \or \pgfmathrandomitem{\b}{CoprimeWithFour}
  \or \pgfmathrandomitem{\b}{CoprimeWithFive} 
 \fi
 \edef\b{\b}


\pgfmathtruncatemacro{\asqr}{\a^2}

\pgfmathtruncatemacro{\bsqr}{\b^2}



\pgfmathdeclarerandomlist{varxy}{{y}{x}{a}{p}{t}}
\pgfmathrandomitem{\varx}{varxy}
\edef\varxy{\varx}









Factor completely the following polynomial.
\[\invisibleoneincoef{\asqr}\varx^2  - \bsqr.\]


\begin{solution}
\[\begin{split}
&\invisibleoneincoef{\asqr}\varx^2  - \bsqr \\
=&(\invisibleoneincoef{\a}\varx)^2  - (\b)^2\\
=&(\invisibleoneincoef{\a}\varx  - \b)(\invisibleoneincoef{\a}\varx  + \b)
\end{split}
\]
\end{solution}
