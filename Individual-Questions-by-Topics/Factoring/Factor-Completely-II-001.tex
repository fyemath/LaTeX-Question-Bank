 \pgfmathtruncatemacro{\f}{random(2,5)}

\pgfmathtruncatemacro{\fB}{\f*\B}
\pgfmathtruncatemacro{\fC}{\f*\C}

Factor completely the following polynomial.
\[\fA \varx^3  \plusminus{\fB}\varx^2 \plusminus{\fC}\varx.\]

\begin{solution}
\begin{multicols}{2}

\[\begin{split}
	& \fA\varx^3  \plusminus{\fB}\varx^2 \plusminus{\fC}\varx\\
  =&\f\varx(\A\varx^2  \plusminus{\B}\varx \ifnum\C<0 \C\else +\C\fi )\\
  =&\f\varx(\invisibleoneincoef{\a}\varx-\b)(\c\varx+\d).
\end{split}
\]
\columnbreak

\begin{center}
  Factor $\A\varx^2  \plusminus{\B}\varx \ifnum\C<0 \C\else +\C\fi$

\end{multicols}

\end{solution}
