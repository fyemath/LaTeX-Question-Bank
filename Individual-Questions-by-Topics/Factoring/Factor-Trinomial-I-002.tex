 \pgfmathtruncatemacro{\A}{\a*\c}
 \pgfmathtruncatemacro{\C}{(-\b)*(\d)}
 \pgfmathtruncatemacro{\bc}{(-\b)*\c}
 \pgfmathtruncatemacro{\ad}{\a*(\d)}
 \pgfmathtruncatemacro{\B}{\ad+\bc}


Factor completely the following polynomial.
\[\A \varx^2  \plusminus{\B}\varx \plusminus{\C}.\]

\begin{solution}
\begin{multicols}{2}

\[
	\A\varx^2  \plusminus{\B}\varx \plusminus{\C}=(\invisibleoneincoef{\a}\varx-\b)(\c\varx+\d).
\]
\columnbreak

\begin{tikzpicture}
 \matrix (m) [
 matrix of math nodes,
row sep=-\pgflinewidth,
 column sep=-.5\pgflinewidth,
 minimum width=2em,
 ]
 { A=\A &{}&C=\C & \\
   \a &{}&-\b & \\
   \c &{}&\d &  \\
    \bc& + & \ad &=\B=B\\
 };
 \path
   (m-2-1) edge (m-3-3)
   (m-2-3) edge (m-3-1); 
 \draw[red] (m-4-1.north west) -- (m-4-4.north east);  
\end{tikzpicture}
\end{multicols}

\end{solution}
