\pgfmathtruncatemacro{\a}{-random(1,3)} 
\pgfmathtruncatemacro{\b}{-random(1,3)} 

\pgfmathtruncatemacro{\x}{random(3,8)} 
\pgfmathtruncatemacro{\y}{random(2,5)}

\pgfmathtruncatemacro{\mnum}{\y-\b}

\pgfmathtruncatemacro{\mden}{\x-\a} 

\pgfmathtruncatemacro{\mrnum}{\mnum/gcd(\mnum,\mden)} 
\pgfmathtruncatemacro{\mrden}{\mden/gcd(\mnum,\mden)} 




Find the slope of the line passing through the following two points
\[(\a, \b)\text{\quad and\quad} (\x, \y).\]

\begin{solution}
The slope is
\[
m=\frac{\y-(\b)}{\x-(\a)}=\frac{\mrnum}{\mrden}.
\]
\end{solution}
