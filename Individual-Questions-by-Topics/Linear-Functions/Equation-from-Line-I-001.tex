\pgfmathtruncatemacro{\aa}{random(1,3)} 
\pgfmathtruncatemacro{\bb}{random(3,1)}
\pgfmathtruncatemacro{\cc}{random(1,7)-4}

\pgfmathtruncatemacro{\mtop}{-\aa*random(1,2)}


\pgfmathsetmacro{\mm}{\mtop/\bb}


\pgfmathtruncatemacro{\xx}{\bb}
\pgfmathtruncatemacro{\yy}{\mtop+\cc}


Find the slope-intercept form equation of the line in the following graph.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
	grid=both,
	minor tick num=1,
	xmax=6,
	xmin=-6,
	ymax=6,
	ymin=-6,
]
\addplot [
    domain=-6:6, 
    samples=100, 
    color=blue,
    ]
	{\mtop/\bb*x +\cc};
	\draw[fill=black] (0,\cc) circle (2pt) ;
	\draw[fill=black]  (\xx,\yy) circle (2pt) ;
\end{axis}
\end{tikzpicture}
\end{center}

\begin{solution}
Since the $y$-intercept is $(0, \cc)$ and the point $(\xx, \yy)$ is also on the line, the slope of the line is  
\[
m=\frac{\yy-\display{\cc}}{\xx-0}=\rdfrac{\mtop}{\bb}.
\]

Then the slope-intercept form equation of the line is
\[
	y=\rdfrac{\mtop}{\bb} x \constdisplay{\cc}.
\]
\end{solution}
